import numpy as np
def gaussian_kl(mu1, sigma1, mu2, sigma2):
    d = sigma2 / sigma1
    kl = np.log(d) + (mu1-mu2)**2 / (2*sigma2**2) + 0.5*(1./d)**2 - 0.5
    return kl
